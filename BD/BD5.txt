SELECT ENAME
FROM EMP,DEPT 
WHERE EMP.DEPTNO = DEPT.DEPTNO AND DNAME = 'SALES';

SELECT EMPNO,ENAME
FROM EMP
WHERE COMM > (SELECT AVG(COMM) FROM EMP);

SELECT DNAME,SUM(SAL) 
FROM EMP,DEPT
WHERE EMP.DEPTNO = DEPT.DEPTNO
GROUP BY DNAME;

SELECT ENAME 
FROM EMP 
WHERE EXTRACT(YEAR FROM HIREDATE) = (SELECT EXTRACT(YEAR FROM HIREDATE) FROM EMP WHERE ENAME='SMITH');

SELECT DNAME,COUNT(ENAME)
FROM EMP,DEPT
WHERE EMP.DEPTNO = DEPT.DEPTNO
GROUP BY DNAME
HAVING COUNT(ENAME) > 2;

SELECT DNAME,MAX(SAL)
FROM EMP,DEPT
WHERE EMP.DEPTNO = DEPT.DEPTNO
GROUP BY DNAME;

SELECT ENAME 
FROM EMP
WHERE COMM IS NULL;

SELECT DNAME,ROUND(AVG(SAL),2) AS SREDNIA 
FROM EMP
RIGHT JOIN DEPT
ON EMP.DEPTNO = DEPT.DEPTNO
GROUP BY DNAME;
